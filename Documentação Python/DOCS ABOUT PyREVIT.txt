def modify_document(doc):
    try:
        # Iniciando uma transação
        t = Transaction(doc, "Modificar Documento")
        t.Start()

        # Código que modifica o documento vai aqui
        # Exemplo: modificação de elementos, criação de paredes, etc.

        # Commit para finalizar a transação com sucesso
        t.Commit()

    except Exception as e:
        # Caso ocorra qualquer erro, rollback na transação
        print(f"Erro: {e}")
        if t.HasStarted():
            t.RollBack()

# Verificar se o documento pode ser modificado
if not doc.IsReadOnly and doc.IsModifiable:
    modify_document(doc)
else:
    print("O documento está em modo de somente leitura ou não pode ser modificado.")
